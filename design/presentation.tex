% =======================================================

\begin{frame}{Motivating Example: Magic State Cultivation (MSC)}
  \textbf{The Goal:} Evaluate Magic State Cultivation protocols over $10^{12}$ shots.
  \vspace{0.8em}

  \begin{columns}[T,onlytextwidth]
    \begin{column}{0.52\textwidth}
      \centering
      \includegraphics[width=\textwidth]{msc-overview.png}
    \end{column}

    \begin{column}{0.44\textwidth}
      \begin{alertblock}{The Challenge}
        {\small
        \begin{itemize}\itemsep0.4em
          \item Cultivation includes non-Clifford $T$ gates, so Stim does not apply.
          \item End-to-end 463-qubit cultivation is intractable with existing simulators:
            \begin{itemize}\itemsep0.25em
              \item \textbf{Statevector (Qiskit):} \alert{$\mathcal{O}(2^N)$}
              \item \textbf{Stabilizer rank (tsim):} \alert{$\mathcal{O}(2^{0.4t})$}
              \item \textbf{Generalized stabilizers (SOFT):} heavy \alert{$\mathcal{O}(n^2)$} per-shot overhead
            \end{itemize}
        \end{itemize}
        }
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The Approach: Making Simulation Feasible}
  To simulate early fault-tolerant circuits at scale, we must avoid the exponential walls of standard exact simulators.

  \vspace{1em}
  \textbf{The Core Insight:}
  \begin{itemize}
      \item We break the exponential bottlenecks by cleanly decoupling deterministic routing from probabilistic non-Clifford entanglement.
      \item By tracking these separately, the simulation scales exponentially \textit{only} with the \textbf{Peak Active Rank} ($k_{\max}$), bypassing the physical qubit ($N$) and T-count ($t$) walls entirely.
      \item But beyond SOFT, we do the $k_{\max}$ scaling \textit{without} incurring the \alert{$\mathcal{O}(n^2)$} per-shot overhead.
  \end{itemize}
\end{frame}

\begin{frame}{The ``New'' UCC: Driven by the Math}
  This decoupled mathematical approach demands a fundamentally different software architecture. It suggests a split design to handle the deterministic and probabilistic regimes separately.
  \vspace{1em}

  \textbf{Previous UCC:} An open-source routing and bridging compiler. \\
  \vspace{0.5em}
  \textbf{New UCC:} A \textbf{Multi-Level Compiler} and \textbf{Virtual Machine (VM)} built explicitly for the Early FTQC Era.

  \vspace{1em}
  \begin{itemize}
      \item \textbf{The Compiler (Offline):} Deterministically resolves and absorbs the Clifford routing overhead.
      \item \textbf{The VM (Online):} Executes the isolated probabilistic non-Clifford operations at runtime.
  \end{itemize}
\end{frame}

\section{The Core Theory: The Factored State}
% =======================================================

\begin{frame}{Circuit Composition \& The Heisenberg Mapping}
  We partition all quantum operations into two sets:
  \begin{enumerate}
      \item \textbf{Deterministic:} Clifford Unitaries ($CNOT, H, \ldots$).
      \item \textbf{Active:} Non-Clifford rotations ($T$), Pauli Measurements, Pauli Noise, Conditional Paulis.
  \end{enumerate}

  \vspace{1em}

  This covers a universal gateset, most common noise models and feed-forward mid-circuit measurement.

  It doesn't cover conditional non-clifford gates.
\end{frame}

\begin{frame}{The Factored State Equation}
  We define the exact physical quantum state at time $t$ via a strict factorization:
  \begin{equation*}
      \Large |\psi^{(t)}\rangle = \gamma^{(t)} U_C^{(t)} \tilde{P}^{(t)} \Big( |\phi^{(t)}\rangle_A \otimes |0\rangle_D \Big)
  \end{equation*}

  \vspace{0.2em}
  \begin{itemize}
      \item \textbf{$U_C^{(t)}$ (Clifford Frame):} Deterministic geometric coordinate mapping.
      \item \textbf{$\tilde{P}^{(t)}$ (Virtual Pauli Frame):} Tracks discrete stochastic errors via virtual $X$ and $Z$ vectors: $\tilde{P}^{(t)} = \bigotimes_{j=0}^{n-1} X_j^{x_j^{(t)}} Z_j^{z_j^{(t)}}$. Evaluated via $\mathcal{O}(1)$ bitwise XORs.
      \item \textbf{$|\phi^{(t)}\rangle_A$ (Active Statevector):} A dense $2^{\mathbf{k}}$ array tracking non-Clifford superposition on $k$ \textit{Active} virtual qubits.
      \item \textbf{$|0\rangle_D$ (Dormant Qubits):} The $n-k$ qubits mathematically locked in $|0\rangle$.
      \item \textbf{$\gamma^{(t)}$ (Global Scalar):} Tracks continuous phases and deferred normalization.
  \end{itemize}
\end{frame}

\begin{frame}{What happens?}

  \begin{equation*}
      \Large |\psi^{(t)}\rangle = \gamma^{(t)} U_C^{(t)} \tilde{P}^{(t)} \Big( |\phi^{(t)}\rangle_A \otimes |0\rangle_D \Big)
  \end{equation*}

  If a gate in the circuit is a Clifford $C$, we simply absorb it -- $U_C^{(t+1)} = C \cdot U_C^{(t)}$.

  If a gate is an "active" operation $O$, its physical Pauli maps to a virtual Pauli via clifford rules. Thanks stim tableau!

  \begin{equation*}
      P U_C^{(t)} = U_C^{(t)} \tilde{P} \quad \text{(Pauli Mapping)}
  \end{equation*}
\end{frame}

\begin{frame}{Virtual Compression (The ``Secret Sauce'')}
  $\tilde{P}$ can touch many virtual qubits (e.g., $\tilde{P} = X_0 \otimes Z_1$). How do we apply it without evaluating a full $2^n$ statevector?

  \vspace{0.5em}
  \begin{block}{Lemma 1: Exact Pauli Compression}
      The Multi-Level compiler constructively synthesizes a sequence of virtual Cliffords $V$ to compress \textit{any} multi-qubit Pauli onto a \textbf{single virtual axis} $v$. ($V \tilde{P} V^\dagger = P_v$).
  \end{block}

  \begin{block}{Lemma 2: Zero-Cost Dormancy}
      If $V$ uses a Dormant qubit as a control (e.g., CNOT$_{D \to A}$), it mathematically acts as the Identity on the Active statevector. \textbf{Zero FLOPs required.}
  \end{block}
\end{frame}

\begin{frame}{Applying the Compression: Where does $V$ go?}
  If we apply virtual Cliffords $V$ to compress an operator $O(\tilde{P})$, how is the global state preserved?

  \vspace{0.5em}
  The compiler mathematically inserts the identity $V^\dagger V = I$ into the factored state, effectively \textbf{pushing it forward}:
  \begin{equation*}
      \dots \underbrace{U_C^{(t)} \alert{V^\dagger}}_{\text{1. Offline Frame}} \;\; \underbrace{\Big[ \alert{V} O(\tilde{P}) \alert{V^\dagger} \Big]}_{\text{2. Localized Op}} \;\; \underbrace{\Big( \alert{V} \tilde{P}^{(t)} \alert{V^\dagger} \Big)}_{\text{3. Pauli Tracker}} \;\; \underbrace{\alert{V} \Big( |\phi^{(t)}\rangle_A \otimes |0\rangle_D \Big)}_{\text{4. Active Array}}
  \end{equation*}

  \vspace{0.05em}
  The transformation $V$ distributes exactly where it is needed:
  \begin{enumerate}
      \item \textbf{Clifford Frame:} Absorbs the inverse offline ($U_C \leftarrow U_C V^\dagger$).
      \item \textbf{The Operation:} Compresses to act on a single virtual axis ($O(P_v)$).
      \item \textbf{Pauli Frame:} Is dynamically conjugated via $\mathcal{O}(1)$ bitwise XORs.
      \item \textbf{Active Array:} Undergoes the transformation $V$ (which costs \textbf{zero FLOPs} if controlled by a dormant qubit!).
  \end{enumerate}
\end{frame}

\section{Software Architecture}
% =======================================================

\begin{frame}{The ``Aha!'' Moment: Theorem 1}
  Look closely at the walkthrough.
  When did the \textcolor{blue}{Clifford Frame ($U_C$)} and \textcolor{blue}{active bounds ($A, D$)} update?
  When did the \textcolor{violet}{statevector ($|\phi\rangle_A$)} and \textcolor{violet}{Pauli tracker ($\tilde{P}$)} compute?

  \vspace{1em}
  \begin{alertblock}{Theorem 1: Clifford Frame Determinism}
      The evolution of the \textcolor{blue}{Clifford Frame $U_C$} and the \textcolor{blue}{dimensional bounds of the active set ($k_{\max}$)} depend \textit{exclusively} on the sequence of physical operations. They are strictly independent of \textcolor{violet}{probabilistic measurement outcomes} or \textcolor{violet}{continuous non-Clifford amplitudes}.
  \end{alertblock}

  \vspace{1em}
  \textbf{Consequence:} We do not need to do heavy tableau math online. We cleanly split the simulation into an \textbf{\textcolor{blue}{Offline Multi-Level Compiler}} and an \textbf{\textcolor{violet}{Online Virtual Machine}}.
\end{frame}

\begin{frame}{Decoupling Computational Complexity}
  \textbf{1. Multi-Level Compiler (Offline, once per circuit):}
  \begin{itemize}
      \item Requires the $\mathcal{O}(G n^2)$ topological matrix tracking.
      \item Evaluates optimal Pauli Compressions to single virtual axes.
      \item Computes global algebraic optimizations over the timeless \textbf{Heisenberg IR (HIR)}.
  \end{itemize}

  \vspace{1em}
  \textbf{2. The Schrödinger Virtual Machine (Online, once per shot):}
  \begin{itemize}
      \item Clifford is just bit operations
      \item Non-clifford bounded $\mathcal{O}(2^{k_{\max}})$.
      \item No dynamic memory allocation, no tableau tracking. Executes purely localized RISC bytecode.
      \item \textit{Note: C++ engineering making the VM fast (32-byte opcodes, branchless hardware popcounts) is a topic for a future deep-dive!}
  \end{itemize}
\end{frame}
