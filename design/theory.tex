\begin{abstract}
\noindent This document establishes the formal framework for simulating Clifford+T circuits using generalized stabilizer states. By extending the reference frame and parity tracking approach of Stim, we introduce a theoretical partition between geometric basis transformations and probabilistic state evolution. We prove that the superposition dimension and basis alignment of the active state can be evaluated ahead of time, independent of probabilistic runtime collapse. This natural decoupling motivates an offline-compilation and fast-sampling architecture (e.g., the HIR compiler and SchrodingerVM of UCC), where heavy geometric basis alignments are evaluated offline, allowing the complex probability amplitudes to execute over minimal, dense state arrays using optimized division-free operations.
\end{abstract}

\section{The Circuit Model}
\label{sec:circuit_model}
A quantum circuit over $n$ qubits is a series of operations $O^{(t)}$ applied at discrete time steps $t=0,\ldots$. At $t=0$ qubits are in the computational basis state $\ket{0}^{\otimes n}$. Let $P \in \mathcal{P}_n$ denote physical $n$-qubit Pauli string operators, with $\mathcal{P}_n = \{\pm 1, \pm i\}\{I, X, Y, Z\}^{\otimes n}$ representing the $n$-qubit Pauli group.

Each operation is one of the following types:
\begin{itemize}
    \item \textbf{Clifford Unitaries ($U$):} $U \in \mathcal{C}_n$. The Clifford group is $\mathcal{C}_n = \{U \in U(2^n) : U \mathcal{P}_n U^\dagger = \mathcal{P}_n\}$---the set of gates that maps Paulis to Paulis. $\{H, S, CNOT\}$ is a common generating set for $\mathcal{C}_n$.
    \item \textbf{Non-Clifford Unitaries (e.g., $T_q$):} $T_q = e^{i\pi/8} \cos(\frac{\pi}{8}) [ I - i \tan(\frac{\pi}{8}) Z_q ]$. We will focus on $T$ although the formalism generalizes to any unitary generated by a Pauli operator.
    \item \textbf{Projective Pauli Measurements ($P$):} We measure $P \in \mathcal{P}_n$. It has outcomes $m \in \{0, 1\}$ corresponding to the observed eigenvalue $(-1)^m$, with projectors $\Pi_m = \frac{1}{2}(I + (-1)^m P)$.
    \item \textbf{Stochastic Pauli Noise ($E$):} $E \in \mathcal{P}_n$ is a Pauli error that occurs with some probability $p_E$.
    \item \textbf{Classically Controlled Paulis (e.g., $P^{m^{(k)}}$):} Applies $P \in \mathcal{P}_n$ conditioned on a classical measurement outcome $m^{(k)}$ recorded at an earlier step $k$.
\end{itemize}
It will be useful to assume that all operations are single-qubit or two-qubit gates, as is common in quantum circuits.

Although there are valid operations outside this set, this model is universal for quantum computing. The remainder of this document focuses on the theoretical framework for simulating circuits composed of these operations.

\section{Stabilizer-Destabilizer Tableaus}

\begin{itemize}
    \item $\mathcal{S} \subset \mathcal{P}_n$ is a stabilizer group which stabilizes the $n$-qubit stabilizer state $\ket{\psi_{\mathcal{S}}}$ with $+1$ eigenvalue. The group does not contain $-I$ to avoid being trivial.
    \item $\mathcal{S} = \langle s_1, \dots, s_n \rangle$ is generated by $n$ independent commuting signed Pauli strings $s_i$, where $s_i \ket{\psi_{\mathcal{S}}} = \ket{\psi_{\mathcal{S}}}$ for all $i$.
    \item $\mathcal{D} = \langle d_1, \dots, d_n \rangle$ is the destabilizer group generated by $n$ independent signed Pauli strings $d_i$ that anticommute with their corresponding stabilizer generators: $\{d_i, s_i\} = 0$ and commute with all others: $[d_i, s_j] = 0$ for $j \neq i$, that is $d_i s_j = (-1)^{\delta_{ij}} s_j d_i$.
    \item In the canonical form, destabilizers mutually commute: $[d_i, d_j] = 0$.
    \item Define the (signed) Pauli $\mathcal{T} = (\mathcal{S}, \mathcal{D})$ as the joint stabilizer/destabilizer pair along with their signs.
    \item $\mathcal{T}$ serves as a basis for the Pauli group, where we define the extraction as a grouped product:
    \begin{equation} \label{eq:pauli_decomposition}
        P = \omega \left(\prod_{i=1}^n d_i^{x_i}\right) \left(\prod_{i=1}^n s_i^{z_i}\right) = \omega \mathbf{d}^\mathbf{x} \mathbf{s}^\mathbf{z}
    \end{equation}
    where the product is grouped such that all destabilizers precede all stabilizers to consistently fix the phase $\omega$, with $\omega \in \{\pm 1, \pm i\}$ and the bitwise vectors $\mathbf{x}, \mathbf{z} \in \mathbb{F}_2^n$\footnote{Throughout the rest of the document, we will use boldface lowercase letters to represent these bitwise vectors for compactness. We define the standard basis vector $\mathbf{e}_i \in \mathbb{F}_2^n$ (where the $i$-th bit is 1 and all others 0), the bitwise dot product $\mathbf{b}_1 \cdot \mathbf{b}_2 = \sum_{i=1}^n b_{1,i} b_{2,i} \pmod 2$, the bitwise XOR $(\mathbf{b}_1 \oplus \mathbf{b}_2)_i = b_{1,i} \oplus b_{2,i} \pmod 2$, the bitwise AND $(\mathbf{b}_1 \wedge \mathbf{b}_2)_i = b_{1,i} \cdot b_{2,i} \pmod 2$, the bitwise OR $(\mathbf{b}_1 \vee \mathbf{b}_2)_i = b_{1,i} \lor b_{2,i}$, and the bitwise NOT $(\neg \mathbf{b})_i = 1 - b_i \pmod 2$. The subset relation $\boldsymbol{\alpha} \subseteq \mathbf{b}$ denotes bitwise inclusion ($\boldsymbol{\alpha} \wedge \neg \mathbf{b} = \mathbf{0}$). We use $\boldsymbol{\alpha}$ to represent indexing sub-masks associated with branches of a superposition.} indicating the presence of the corresponding generators in the decomposition of $P$.
    \item Let $\Gamma(P, \mathcal{T}) \mapsto (\omega, \mathbf{x}, \mathbf{z})$ be the \emph{Pauli Extraction Map} that extracts this decomposition.
    \item Let $\Lambda_C(U, \mathcal{T}) \mapsto \mathcal{T}'$ represent the \emph{Clifford Conjugation} operation that updates the tableau by conjugating all generators by a physical Clifford unitary $U$, i.e. $\mathcal{T}' = (U \mathcal{S} U^\dagger, U \mathcal{D} U^\dagger)$. This is a symplectic transformation, meaning it preserves all canonical commutation relations and rotates the geometry of the tableau.
    \item Let $\Lambda_{\text{rowsum}}(i, j, \mathcal{T}) \mapsto \mathcal{T}'$ represent a \emph{Symplectic Rowsum}. This operation performs a change-of-basis by multiplying source stabilizer $j$ into target stabilizer $i$, replacing $s_i \leftarrow s_i s_j$. To preserve canonical commutation relations (preventing $d_j$ from anticommuting with the new $s_i$), the destabilizer must concurrently update via $d_j \leftarrow d_j d_i$. Because $s_i$ and $s_j$ both stabilize the vacuum state, their product $s_i s_j$ also stabilizes it. This acts as a structural gauge choice to reshape the tableau's geometry without altering the physical vacuum state.
    \item Let $\Lambda_{\text{pivot}}(k_p, \omega', \mathbf{z}', \mathcal{T}) \mapsto \mathcal{T}'$ represent the \emph{Aaronson-Gottesman (AG) Pivot}\footnote{In quantum literature, the overall measurement resolution process is frequently referred to as symplectic Gaussian elimination. Strictly speaking, Gaussian elimination specifically denotes the prerequisite $\mathcal{O}(n^2)$ preparation phase of iterative rowsums, while the AG Pivot is the subsequent $\mathcal{O}(1)$ action of swapping the generators ($s_k \leftrightarrow d_k$) and phase anchoring.} operation that performs a change-of-basis by swapping generators and restoring canonical commutation relations for a new observable $P$.

    This happens post measurement, when the state collapses for a physical measurement $P \in \mathcal{P}_n$. If $P$ commutes with all stabilizers, this is a no-operation. Assuming anticommutativity has already been cleared from all stabilizers except a chosen pivot $s_{k_p}$ (via prerequisite rowsums, which guarantees the extracted $X$-support $\mathbf{x}$ is exactly $\mathbf{e}_{k_p}$), the AG Pivot operation requires the pivot axis $k_p$, the phase $\omega'$, and the current $Z$-support $\mathbf{z}'$. It demotes $s_{k_p}$ to the new destabilizer $d_{k_p}$ and sets the new $s_{k_p}$ to $P$, handling phase housekeeping to maintain canonical form.
    \item Following the insight of Stim, we maintain both a forward tableau $\mathcal{T}_{fwd}$ and an inverse tableau $\mathcal{T}_{inv}$. The forward tableau defines the current generators in terms of physical Paulis, while the inverse tableau defines physical Paulis in the current stabilizer basis. For the local, constant-weight observables typical of quantum circuits, the inverse tableau can evaluate the Pauli extraction map using $\mathcal{O}(n)$ bitwise operations (executed as highly-optimized, contiguous SIMD vector reads) by simply reading and XORing the corresponding rows of $\mathcal{T}_{inv}$. In contrast, extracting this using only the forward tableau would require an $\mathcal{O}(n^2)$ generic inner product against all $2n$ rows, or---if optimized for local observables---an $\mathcal{O}(n)$ sequence of scattered, cache-unfriendly scalar column-extractions that defeats vectorization. Given that the Pauli extraction map is used in every operation that interacts with the physical hardware (measurements, non-Clifford gates, noise), this is a critical optimization.
    \item Clifford conjugation scales as $\mathcal{O}(n)$ vector operations for low-arity gates.

    \item Symplectic Rowsums scale as $\mathcal{O}(n)$ vector operations since one stabilizer and one destabilizer are updated in both tableaus.
    \item The overall AG Pivot process (including the prerequisite Gaussian elimination) scales as $\mathcal{O}(n^2)$ vector operations because it requires manipulating the entire tableau.
\end{itemize}


\section{The Unified State Tuple and Symplectic Determinism}
\label{sec:state_and_symplectic_determinism}

In addition to being a basis for the Pauli group, the tableau $\mathcal{T}$ induces a basis for the entire Hilbert space of $n$-qubits. The destabilizers act as ladder operators that shift the active basis state by toggling the presence of their corresponding operator in the product, while the stabilizers define the vacuum state $\ket{\psi_{\mathcal{S}}}$. To make this perspective, clear, we will switch notation to $\ket{\vacuumS}$ for the state stabilized by $\mathcal{S}$.

We partition the generalized state at time $t$ into a discrete, unified tuple $\mathcal{X}^{(t)}$ composed of two domains: the geometric frame and the probabilistic state.
\begin{equation}
    \mathcal{X}^{(t)} := \Big( \underbrace{\mathcal{T}^{(t)}, \mathbf{b}^{(t)}}_{\text{Geometric State}}, \quad \underbrace{\mathcal{M}^{(t)}, \gamma^{(t)}, \mathbf{p}_X^{(t)}, \mathbf{p}_Z^{(t)}, a^{(t)}}_{\text{Probabilistic State}} \Big)
\end{equation}
where:
\begin{itemize}
    \item $\mathcal{T}^{(t)} \equiv (\mathcal{S}^{(t)}, \mathcal{D}^{(t)})$ is the fully signed geometric frame.
    \item $\mathbf{b}^{(t)} \in \mathbb{F}_2^n$ is the \emph{active support mask}. It isolates which destabilizers are currently active for at least some non-zero terms in the superposition. For all indices where $b_i^{(t)} = 0$, the corresponding generator $d_i$ is dormant.
    \item $\mathcal{M}^{(t)}$ is the classical memory record, defined as an ordered sequence tracking all historical binary measurement outcomes. We use $\langle \rangle$ to denote an empty sequence and $\Vert$ for sequence concatenation.
    \item $\gamma^{(t)} \in \mathbb{C}$ is a global complex scale factor tracking the physical norm and global continuous phase.
    \item $\mathbf{p}_X^{(t)}, \mathbf{p}_Z^{(t)} \in \mathbb{F}_2^n$ are purified classical parity trackers governing the dynamic \textit{probabilistic deviations} (runtime physical signs).
    \item $a^{(t)} \in \mathbb{C}^{2^{|\mathbf{b}^{(t)}|}}$ is the complex amplitude array. It is indexed by the active branch sub-mask $\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}$. Because $\boldsymbol{\alpha}_k = 0$ for all dormant bits $k \notin \mathbf{b}^{(t)}$, the dimension of this array is exactly $2^{|\mathbf{b}^{(t)}|}$. These entries are unnormalized; we rely on $\gamma^{(t)}$ to track the global norm and global phase, while $a^{(t)}$ tracks the relative complex structure of the superposition.
\end{itemize}
We assume $\xi^{(t)} \in [0, 1)$ is a globally available continuous random variable that we can query at any time step $t$ to resolve stochastic outcomes.

For our initial state, we have $\mathcal{T}^{(0)}$ defined by the $n$-qubit computational basis state, $\mathbf{b}^{(0)} = \mathbf{0}$, $\mathcal{M}^{(0)} = \langle \rangle$, $\gamma^{(0)} = 1$, $\mathbf{p}_X^{(0)} = \mathbf{0}$, $\mathbf{p}_Z^{(0)} = \mathbf{0}$, and $a^{(0)} = [1]$.

The goal of this section is to formalize how these state variables change for each circuit operation. We will find that the geometric state rules are independent of the probabilistic state, enabling the decoupling of offline geometric compilation from online probability sampling.

\subsection{The Physical State Equation}
To recover the physical quantum state $\ket{\psi_{\mathcal{X}^{(t)}}}$ from the unified tuple $\mathcal{X}^{(t)}$, we define the \textit{true physical generators} by attaching the runtime probabilistic parities to the geometric generators:
\begin{equation}
    \bar{d}_i^{(t)} = (-1)^{p_{X,i}^{(t)}} d_i^{(t)} \quad \text{and} \quad \bar{s}_i^{(t)} = (-1)^{p_{Z,i}^{(t)}} s_i^{(t)}
\end{equation}

The physical base state/vacuum state $\ket{\vacuumSPZt}$ is defined as the simultaneous $+1$ eigenstate of the physical stabilizers. The full physical state is then constructed by applying the complex amplitudes over the active sub-basis defined by the mask $\mathbf{b}^{(t)}$ (summing over bitstrings $\boldsymbol{\alpha}$ where $\boldsymbol{\alpha}_i = 0$ for all $b_i^{(t)} = 0$):
\begin{equation} \label{eq:physical_state_expanded}
    \ket{\psi_{\mathcal{X}^{(t)}}} = \gamma^{(t)} \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} a_{\boldsymbol{\alpha}}^{(t)} (\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha}} \ket{\vacuumSPZt}
\end{equation}
where we use the vector exponentiation notation $(\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha}} \equiv \prod_{i=1}^n (\bar{d}_i^{(t)})^{\boldsymbol{\alpha}_i}$. In some derivations, we will expand the physical generators which yields $(\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha}} = (-1)^{\mathbf{p}_X^{(t)} \cdot \boldsymbol{\alpha}} (\mathbf{d}^{(t)})^{\boldsymbol{\alpha}}$.

\subsection{Tableau Operations on the State}
\subsubsection{Pauli Action}
\label{sec:pauli_action}
For all operations aside from Clifford gates, we will need to consider the action of a localized pauli $P^{(t)}$ action on the current state. From the Pauli extraction map $\Gamma(P^{(t)}, \mathcal{T}^{(t)}) \mapsto (\omega, \mathbf{x}, \mathbf{z})$, which means we can write

\begin{equation}
P^{(t)} = \omega (\mathbf{d}^{(t)})^\mathbf{x} (\mathbf{s}^{(t)})^\mathbf{z} = \omega (-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)} + \mathbf{z} \cdot \mathbf{p}_Z^{(t)}} (\bar{\mathbf{d}}^{(t)})^\mathbf{x} (\bar{\mathbf{s}}^{(t)})^\mathbf{z}
\end{equation}
where we've expanded using the physical stabilizers and destabilizers.

Applying this to Eq.~\ref{eq:physical_state_expanded} yields:
\begin{align}
    P^{(t)} \ket{\psi_{\mathcal{X}^{(t)}}} &= \gamma^{(t)} \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} a_{\boldsymbol{\alpha}}^{(t)} \Big( \omega (-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)} + \mathbf{z} \cdot \mathbf{p}_Z^{(t)}} (\bar{\mathbf{d}}^{(t)})^\mathbf{x} (\bar{\mathbf{s}}^{(t)})^\mathbf{z} \Big) (\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha}} \ket{\vacuumSPZt} \notag \\
    &= \gamma^{(t)} \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} a_{\boldsymbol{\alpha}}^{(t)} \omega (-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)} + \mathbf{z} \cdot \mathbf{p}_Z^{(t)}} (-1)^{\mathbf{z} \cdot \boldsymbol{\alpha}} (\bar{\mathbf{d}}^{(t)})^\mathbf{x} (\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha}} (\bar{\mathbf{s}}^{(t)})^\mathbf{z} \ket{\vacuumSPZt} \quad \text{(Commute } \bar{\mathbf{s}} \text{ past } \bar{\mathbf{d}} \text{)} \notag \\
    &= \label{eq:pauli_action} \gamma^{(t)} \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} a_{\boldsymbol{\alpha}}^{(t)} \underbrace{\omega (-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)} + \mathbf{z} \cdot \mathbf{p}_Z^{(t)} + \mathbf{z} \cdot \boldsymbol{\alpha}}}_{\text{Dynamic Phase: } \Phi(\boldsymbol{\alpha}; \omega, \mathbf{x}, \mathbf{z}, \mathbf{p}_X^{(t)}, \mathbf{p}_Z^{(t)})} \Big( (\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha} \oplus \mathbf{x}} \ket{\vacuumSPZt} \Big) \quad \text{(Evaluate } \bar{\mathbf{s}} \text{ on vacuum)}
\end{align}

where we have defined the dynamic phase as
\begin{equation}
    \label{eq:dynamic_phase}
    \Phi(\boldsymbol{\alpha}; \omega, \mathbf{x}, \mathbf{z}, \mathbf{p}_X^{(t)}, \mathbf{p}_Z^{(t)}) = \omega (-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)} + \mathbf{z} \cdot \mathbf{p}_Z^{(t)} + \mathbf{z} \cdot \boldsymbol{\alpha}}
\end{equation}

\subsubsection{Rowsums and State Evolution}
\label{sec:rowsums_and_state_evolution}
When applying a Symplectic Rowsum $\Lambda_{\text{rowsum}}(i, j)$, we reshape the geometry by multiplying source stabilizer $j$ into target stabilizer $i$. To preserve canonical commutativity, the corresponding destabilizers must update inversely.

Let the tilde ($\sim$) denote variables within the \textit{new} mapped geometric basis. The rowsum enforces the following transformation:
\begin{align*}
    \tilde{s}_i &= s_i s_j \quad \implies \quad s_i = \tilde{s}_i \tilde{s}_j \\
    \tilde{d}_j &= d_j d_i \quad \implies \quad d_j = \tilde{d}_j \tilde{d}_i
\end{align*}
All other generators remain unchanged.

\vspace{0.1cm}
\noindent \textbf{Evolution of the Physical State} \\
The physical quantum state $\ket{\psi_{\mathcal{X}^{(t)}}}$ (from Eq.~\ref{eq:physical_state_expanded}) must remain invariant beneath this mutating geometry. We derive the new state representation by substituting the inverted basis relationships into the state equation.

First, we determine the new runtime $Z$-parities that define the new physical vacuum state $\ket{\vacuumStilde}$. Because $\tilde{s}_i = s_i s_j$, the new physical stabilizer inherits the product of the old eigenvalues:
\begin{equation}
    (-1)^{\tilde{p}_{Z,i}} = (-1)^{p_{Z,i}} (-1)^{p_{Z,j}} = (-1)^{p_{Z,i} \oplus p_{Z,j}}
\end{equation}

Next, we substitute the inverted destabilizer relationship $d_j = \tilde{d}_j \tilde{d}_i$ into the geometric product $(\mathbf{d})^{\boldsymbol{\alpha}}$ for a given active branch $\boldsymbol{\alpha}$:
\begin{align*}
    (d_i)^{\boldsymbol{\alpha}_i} (d_j)^{\boldsymbol{\alpha}_j} &= (\tilde{d}_i)^{\boldsymbol{\alpha}_i} (\tilde{d}_j \tilde{d}_i)^{\boldsymbol{\alpha}_j} = (\tilde{d}_i)^{\boldsymbol{\alpha}_i \oplus \boldsymbol{\alpha}_j} (\tilde{d}_j)^{\boldsymbol{\alpha}_j}
\end{align*}
This reveals that the old basis configuration $\boldsymbol{\alpha}$ maps to a new configuration $\tilde{\boldsymbol{\alpha}}$ where the $i$-th bit has been flipped by the $j$-th bit: $\tilde{\boldsymbol{\alpha}}_i = \boldsymbol{\alpha}_i \oplus \boldsymbol{\alpha}_j$, which we write as $\tilde{\boldsymbol{\alpha}} = \boldsymbol{\alpha} \oplus (\boldsymbol{\alpha}_j \cdot \mathbf{e}_i)$. Because this coordinate shift evaluates as an involution over $\mathbb{F}_2$ (applying it twice yields the original index), the inverse mapping is identical: $\boldsymbol{\alpha} = \tilde{\boldsymbol{\alpha}} \oplus (\tilde{\boldsymbol{\alpha}}_j \cdot \mathbf{e}_i)$.

Finally, we substitute this index shift into the runtime $X$-parity phase term $(-1)^{\mathbf{p}_X \cdot \boldsymbol{\alpha}}$ to isolate the new tracking vector $\tilde{\mathbf{p}}_X$:
\begin{align*}
    \mathbf{p}_X \cdot \boldsymbol{\alpha} &= \mathbf{p}_X \cdot \big( \tilde{\boldsymbol{\alpha}} \oplus (\tilde{\boldsymbol{\alpha}}_j \cdot \mathbf{e}_i) \big) = (\mathbf{p}_X \cdot \tilde{\boldsymbol{\alpha}}) \oplus (p_{X,i} \cdot \tilde{\boldsymbol{\alpha}}_j)
\end{align*}
Grouping the terms acting on $\tilde{\boldsymbol{\alpha}}_j$ yields the new $X$-parity: $\tilde{p}_{X,j} = p_{X,j} \oplus p_{X,i}$.

\vspace{0.1cm}
\noindent \textbf{Explicit Update Rules ($\mathcal{X} \mapsto \tilde{\mathcal{X}}$):} \\
To ensure the state remains invariant, the complex amplitudes and physical parity trackers must absorb these coordinate mappings. A geometric rowsum forces the following state tuple update:
\begin{align}
    \tilde{\mathcal{T}} &= \Lambda_{\text{rowsum}}(i, j, \mathcal{T}) \quad &(\text{Geometric Update}) \nonumber \\
    \tilde{\mathbf{b}} &= \mathbf{b}, \quad \tilde{\gamma} = \gamma \quad &(\text{Invariants}) \nonumber \\
    \tilde{p}_{Z, i} &= p_{Z, i} \oplus p_{Z, j} \quad &(\text{Vacuum Alignment}) \nonumber \\
    \tilde{p}_{X, j} &= p_{X, j} \oplus p_{X, i} \quad &(\text{Phase Alignment}) \nonumber \\
    \tilde{a}_{\boldsymbol{\beta}} &= a_{\boldsymbol{\beta} \oplus (\boldsymbol{\beta}_j \cdot \mathbf{e}_i)} \quad \forall \boldsymbol{\beta} \subseteq \tilde{\mathbf{b}} \quad &(\text{Amplitude Reindexing Map}) \label{eq:reindexing}
\end{align}
(Note: we introduce the dummy index $\boldsymbol{\beta}$ to represent indices within the newly mapped array $\tilde{a}$).

\vspace{0.1cm}
\noindent \textbf{The Zero-Cost Dormant Property:}\\
A critical algorithmic optimization emerges from the reindexing map if the source axis of the rowsum is dormant ($j \notin \mathbf{b}^{(t)}$). Because the active support mask dictates that $\boldsymbol{\beta}_j = 0$ across all valid branches of the superposition, the coordinate shift evaluates to zero: $\boldsymbol{\beta} \oplus \mathbf{0} = \boldsymbol{\beta}$. When evaluating a rowsum from a dormant source, the complex array avoids the Amplitude Reindexing Map and passes through untouched. Note that the parity shifts ($\tilde{\mathbf{p}}_Z, \tilde{\mathbf{p}}_X$) must still be dynamically evaluated to preserve the canonical vacuum phase.

\section{Unified State Evolution ($\mathcal{X}^{(t)} \mapsto \mathcal{X}^{(t+1)}$)}

Every circuit operation $O^{(t)}$ maps $\mathcal{X}^{(t)} \mapsto \mathcal{X}^{(t+1)}$. In this section, we derive the explicit update rules for each operation type. If a state member is not listed in a transition rule, it remains invariant.

\subsection{Cliffords}
\textbf{Derivation:} From Clifford Conjugation, $U$ maps the tableau $\mathcal{T}^{(t)}$ to $\mathcal{T}' = \Lambda_C(U, \mathcal{T}^{(t)})$ by conjugating all generators. The vacuum eigenstate condition satisfies the new base state $\ket{\vacuumSPZtnext}$:
\begin{align*}
    \bar{s}_i^{(t+1)} \left( U \ket{\vacuumSPZt} \right) &= U \bar{s}_i^{(t)} U^\dag U \ket{\vacuumSPZt} \\
    &= U \big( \bar{s}_i^{(t)} \ket{\vacuumSPZt} \big) \\
    &= (+1) \left( U \ket{\vacuumSPZt} \right)
\end{align*}

Because symplectic tableau conjugation drops the global phase of $U$, we explicitly accumulate the global phase scalar $\theta_U$ acquired during the operation. Applying $U$ to Eq.~\ref{eq:physical_state_expanded} yields:
\begin{align*}
U \ket{\psi_{\mathcal{X}^{(t)}}} &= (\gamma^{(t)} \cdot e^{i\theta_U}) \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} a_{\boldsymbol{\alpha}}^{(t)} \, (\bar{\mathbf{d}}^{(t+1)})^{\boldsymbol{\alpha}}  \ket{\vacuumSPZtnext}
\end{align*}


\vspace{0.1cm}
\noindent \textbf{Explicit Update Rules ($\mathcal{X}^{(t)} \mapsto \mathcal{X}^{(t+1)}$):}
\begin{align*}
    \mathcal{T}^{(t+1)} &= \Lambda_C(U^{(t)}, \mathcal{T}^{(t)}) \quad &(\text{Clifford Conjugation}) \\
    \gamma^{(t+1)} &= \gamma^{(t)} \cdot e^{i\theta_U} \quad &(\text{Global Phase Accumulation})
\end{align*}
The probabilistic state components ($\mathcal{M}, \mathbf{p}_X, \mathbf{p}_Z, a$) remain invariant.

\subsection{Non-Cliffords (e.g., $T$)}
We evaluate non-Clifford unitaries expanded as sums over Pauli terms, focusing primarily on the $T_q$ rotation: $T_q = e^{i\pi/8} \cos(\frac{\pi}{8}) [ I - i \tan(\frac{\pi}{8}) Z_q]$ which operations on qubit $q$.  The overall phase will get absorbed into $\gamma$ and the $I$ acts trivially, so our main task is to determine how $Z_q$ acts.

\textbf{Derivation:} From the Pauli extraction map $\Gamma(Z_q^{(t)}, \mathcal{T}^{(t)}) \mapsto (\omega, \mathbf{x}, \mathbf{z})$, we geometrically expand the operator as $Z_q^{(t)} = \omega (\mathbf{d}^{(t)})^\mathbf{x} (\mathbf{s}^{(t)})^\mathbf{z}$. Let $\mathbf{x}_{\text{dorm}} = \mathbf{x} \wedge \neg \mathbf{b}^{(t)}$ be the mask of dormant \textit{destabilizers} this operator has support on.

If there is more than one such dormant destabilizer, we isolate this support to a single geometric axis. We select a pivot $k_p \in \operatorname{supp}(\mathbf{x}_{\text{dorm}})$ and iteratively apply the Symplectic Rowsum $\Lambda_{\text{rowsum}}(i, k_p)$ for all other $i \in \operatorname{supp}(\mathbf{x}_{\text{dorm}}) \setminus \{k_p\}$. To understand why this clears the other dormant supports from $Z_q$, consider a case where Pauli extraction maps the product $d_i d_{k_p}$. Applying $\Lambda_{\text{rowsum}}(i, k_p)$ updates the geometry to $\tilde{d}_{k_p} = d_{k_p} d_i$, which inverts to $d_{k_p} = \tilde{d}_{k_p} \tilde{d}_i$ and $d_i = \tilde{d}_i$. Substituting these inverse maps into the operator yields:
\begin{equation*}
    d_i d_{k_p} \mapsto \tilde{d}_i (\tilde{d}_{k_p} \tilde{d}_i) = \tilde{d}_{k_p} (\tilde{d}_i)^2 = \tilde{d}_{k_p}
\end{equation*}
Because canonical destabilizers mutually commute and square to the identity over $\mathbb{F}_2$, the $i$-th destabilizer is eliminated from the geometric expansion. From Section~\ref{sec:rowsums_and_state_evolution}, this intermediate geometric compression preserves the complex array invariant without reindexing $a$ (the Zero-Cost Dormant Property).

Let $(\mathcal{T}_{\text{tmp}}, \mathbf{p}_X^{\text{tmp}}, \mathbf{p}_Z^{\text{tmp}})$ represent the intermediate geometric state and parities after these rowsums. We re-extract the expansion of $Z_q$ in this new intermediate basis: $(\omega', \mathbf{x}', \mathbf{z}') = \Gamma(Z_q^{(t)}, \mathcal{T}_{\text{tmp}})$, restricting support to at most one dormant destabilizer. \subsection{Projective Pauli Measurements ($P^{(t)}$)}

\begin{equation}
    P_0 = p_0 \cdot |\gamma^{(t)}|^2
\end{equation}
Following a measurement collapse, we preserve this overall state norm ($|\gamma|^2 \sum |a|^2 = 1$) performing division exclusively on $\gamma$.

\textbf{Derivation:}

The absolute probability of observing $m=0$ ($+1$ eigenvalue) is governed by the expectation value $P_0 = \frac{1}{2} (1 + \langle \psi_{\mathcal{X}^{(t)}} | P^{(t)} | \psi_{\mathcal{X}^{(t)}} \rangle)$. Because physical destabilizers ($\bar{d}_i^\dagger = \bar{d}_i$) are orthogonal over the vacuum, the inner product evaluates to a Kronecker delta over $\mathbb{F}_2$:
\begin{align*}
    \langle P^{(t)} \rangle &= |\gamma^{(t)}|^2 \sum_{\boldsymbol{\beta} \subseteq \mathbf{b}^{(t)}} \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} \big(a_{\boldsymbol{\beta}}^{(t)}\big)^* a_{\boldsymbol{\alpha}}^{(t)} \Phi(\boldsymbol{\alpha}; \omega, \mathbf{x}, \mathbf{z}, \mathbf{p}_X^{(t)}, \mathbf{p}_Z^{(t)}) \underbrace{\Big\langle \vacuumSPZt \Big| (\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\beta} \oplus \boldsymbol{\alpha} \oplus \mathbf{x}} \Big| \vacuumSPZt \Big\rangle}_{\delta_{\boldsymbol{\beta} \oplus \boldsymbol{\alpha} \oplus \mathbf{x}, \, \mathbf{0}}}
\end{align*}
This forces $\boldsymbol{\beta} = \boldsymbol{\alpha} \oplus \mathbf{x}$, collapsing the double sum. Because the bra sum restricts $\boldsymbol{\beta} \subseteq \mathbf{b}^{(t)}$, the surviving terms require the shifted branch to exist within the active mask:
\begin{equation*}
    \langle P^{(t)} \rangle = |\gamma^{(t)}|^2 \sum_{\substack{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)} \\ (\boldsymbol{\alpha} \oplus \mathbf{x}) \subseteq \mathbf{b}^{(t)}}} \left(a_{\boldsymbol{\alpha} \oplus \mathbf{x}}^{(t)}\right)^* a_{\boldsymbol{\alpha}}^{(t)} \, \Phi(\boldsymbol{\alpha}; \omega, \mathbf{x}, \mathbf{z}, \mathbf{p}_X^{(t)}, \mathbf{p}_Z^{(t)})

The probablistic outcome determined by how the geometric shift $\mathbf{x}$ interacts with the support mask $\mathbf{b}^{(t)}$. \subsection{Conditional Pauli Operations}
We unify stochastic Pauli noise and classical conditional feedback by defining a generic conditional Pauli operation $P^{(t)}(b)$. The physical operator $P^{(t)}$ applies if a Boolean trigger $b \in \{0, 1\}$ evaluates to 1. Specifically:
\begin{itemize}
    \item \textbf{Stochastic Pauli Noise:} Conditioned on an isolated probability rate $p_E^{(t)}$ queried against the random variable $\xi^{(t)}$: $b = \mathbf{1}_{\xi^{(t)} < p_E^{(t)}}$.
    \item \textbf{Classical Feedback:} Conditioned on a prior classical measurement, the random stream is bypassed and $b$ evaluates to the $k$-th element of the classical memory sequence:
    \begin{equation}
        b = m^{(k)} \quad (\text{where } m^{(k)} \in \mathcal{M}^{(t)})
    \end{equation}
\end{itemize}

\textbf{Derivation:} Assuming the condition evaluates to true ($b=1$), applying $P^{(t)}$ to the state evaluates via the Pauli action derived in Eq.~\ref{eq:pauli_action}:
\begin{equation*}
    P^{(t)} \ket{\psi_{\mathcal{X}^{(t)}}} = \gamma^{(t)} \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} a_{\boldsymbol{\alpha}}^{(t)} \omega (-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)} + \mathbf{z} \cdot \mathbf{p}_Z^{(t)} + \mathbf{z} \cdot \boldsymbol{\alpha}} \Big( (\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha} \oplus \mathbf{x}} \ket{\vacuumSPZt} \Big)
\end{equation*}

We absorb this coordinate transform into the tracking parities without mutating the complex array $a^{(t)}$. Let the updated parities be $\mathbf{p}_Z^{(t+1)} = \mathbf{p}_Z^{(t)} \oplus \mathbf{x}$ and $\mathbf{p}_X^{(t+1)} = \mathbf{p}_X^{(t)} \oplus \mathbf{z}$.

First, shifting the runtime $Z$-parity redefines the physical vacuum state. Because applying destabilizers flips the corresponding stabilizer eigenvalues (e.g $s_i$ on $d_i \ket{0_i}$ now has a -1 eigenvalue), the geometric shift maps the vacuum via the un-signed generators: $\ket{{\mathbf{0}_{\mathcal{S}^{(t+1)}}}(\mathbf{p}_Z^{(t+1)})} = (\mathbf{d}^{(t)})^{\mathbf{x}} \ket{\vacuumSPZt}$. Substituting the physical identity $(\bar{\mathbf{d}}^{(t)})^{\mathbf{x}} = (-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)}} (\mathbf{d}^{(t)})^{\mathbf{x}}$ isolates the new vacuum:
\begin{equation*}
    (\bar{\mathbf{d}}^{(t)})^{\mathbf{x}} \ket{\vacuumSPZt} = (-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)}} \ket{\mathbf{0}_{\mathcal{S}^{(t+1)}}(\mathbf{p}_Z^{(t+1)})}
\end{equation*}

Second, shifting the $X$-parity updates the physical destabilizers via $\bar{d}_i^{(t+1)} = (-1)^{z_i} \bar{d}_i^{(t)}$, which establishes the branch relationship $(\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha}} = (-1)^{\mathbf{z} \cdot \boldsymbol{\alpha}} (\bar{\mathbf{d}}^{(t+1)})^{\boldsymbol{\alpha}}$.

Because physical destabilizers mutually commute, $(\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha} \oplus \mathbf{x}} = (\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha}} (\bar{\mathbf{d}}^{(t)})^{\mathbf{x}}$. Substituting these identities directly into the Pauli action equation yields:
\begin{align*}
    P^{(t)} \ket{\psi_{\mathcal{X}^{(t)}}} &= \gamma^{(t)} \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} a_{\boldsymbol{\alpha}}^{(t)} \omega (-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)} + \mathbf{z} \cdot \mathbf{p}_Z^{(t)} + \mathbf{z} \cdot \boldsymbol{\alpha}} \Big( (\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha}} \underbrace{(-1)^{\mathbf{x} \cdot \mathbf{p}_X^{(t)}} \ket{\mathbf{0}_{\mathcal{S}^{(t+1)}}(\mathbf{p}_Z^{(t+1)})}}_{\text{shifted vacuum}} \Big) \\
    &= \gamma^{(t)} \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} a_{\boldsymbol{\alpha}}^{(t)} \omega (-1)^{\mathbf{z} \cdot \mathbf{p}_Z^{(t)} + \mathbf{z} \cdot \boldsymbol{\alpha}} \Big( \underbrace{(-1)^{\mathbf{z} \cdot \boldsymbol{\alpha}} (\bar{\mathbf{d}}^{(t+1)})^{\boldsymbol{\alpha}}}_{(\bar{\mathbf{d}}^{(t)})^{\boldsymbol{\alpha}}} \ket{\mathbf{0}_{\mathcal{S}^{(t+1)}}(\mathbf{p}_Z^{(t+1)})} \Big) \\
    &= \Big[ \gamma^{(t)} \omega (-1)^{\mathbf{z} \cdot \mathbf{p}_Z^{(t)}} \Big] \sum_{\boldsymbol{\alpha} \subseteq \mathbf{b}^{(t)}} a_{\boldsymbol{\alpha}}^{(t)} (\bar{\mathbf{d}}^{(t+1)})^{\boldsymbol{\alpha}} \ket{\mathbf{0}_{\mathcal{S}^{(t+1)}}(\mathbf{p}_Z^{(t+1)})}
\end{align*}

Because $(-1)^{2(\mathbf{x} \cdot \mathbf{p}_X^{(t)})} = 1$ and $(-1)^{2(\mathbf{z} \cdot \boldsymbol{\alpha})} = 1$, the $X$-parity phase and localized array dynamic phase cancel over $\mathbb{F}_2$. This leaves the complex array $a^{(t)}$ invariant, collapsing the physical coordinate transform into a scalar update on $\gamma^{(t)}$.

To integrate the case where the condition evaluates to false ($b=0$), we patch the Boolean indicator $b$ directly into the algebraic updates.

\subsection{The Algebraic Equivalence of the HIR}

To enable global classical optimization without the computational overhead of propagating a local geometric frame, the compiler maps the input circuit into the Heisenberg Intermediate Representation (HIR). We establish the HIR as a parallel construct by proving that tracking rewound operations is equivalent to evaluating the original circuit.

Let a quantum circuit be an alternating sequence of deterministic Clifford operations and active operations (such as non-Clifford rotations, Pauli measurements, conditional Pauli noise, or classical feedback). Because active operations are not unitary (e.g., projective measurements), we represent the circuit as a composition of quantum maps (or channels). For any contiguous run of uninterrupted Clifford gates, we collapse them into a single equivalent Clifford map $\mathcal{C}_k(\rho) = C_k \rho C_k^\dagger$. While collapsing these blocks abstracts away localized hardware routing information, it preserves the exact logical operation. Let $\mathcal{E}_k$ denote the $k$-th active operation map. The full sequence evaluates as:
\begin{equation}
    \mathcal{O}_{\text{total}} = \mathcal{C}_M \circ \mathcal{E}_M \circ \mathcal{C}_{M-1} \circ \mathcal{E}_{M-1} \dots \circ \mathcal{C}_1 \circ \mathcal{E}_1 \circ \mathcal{C}_0
\end{equation}

We define the cumulative Clifford map $\mathcal{U}_k$ as the total deterministic geometric transformation applied from the beginning of the circuit up to step $k$:
\begin{equation}
    \mathcal{U}_k = \mathcal{C}_k \circ \mathcal{C}_{k-1} \dots \circ \mathcal{C}_0
\end{equation}
It follows that $\mathcal{U}_0 = \mathcal{C}_0$. Because Clifford maps are automorphisms generated by unitary matrices, they are invertible. Any local Clifford block at step $k > 0$ can be factored exactly as the transformation between two cumulative frames: $\mathcal{C}_k = \mathcal{U}_k \circ \mathcal{U}_{k-1}^{-1}$. Substituting this identity back into the circuit yields a telescoping sequence:
\begin{equation}
    \mathcal{O}_{\text{total}} = \mathcal{U}_M \circ (\mathcal{U}_{M-1}^{-1} \circ \mathcal{E}_M \circ \mathcal{U}_{M-1}) \circ \dots \circ (\mathcal{U}_0^{-1} \circ \mathcal{E}_1 \circ \mathcal{U}_0)
\end{equation}

Defining the Heisenberg-rewound active operation map as $\tilde{\mathcal{E}}_k := \mathcal{U}_{k-1}^{-1} \circ \mathcal{E}_k \circ \mathcal{U}_{k-1}$, the total circuit operator simplifies to a flat sequence:
\begin{equation}
    \mathcal{O}_{\text{total}} = \mathcal{U}_M \circ \tilde{\mathcal{E}}_M \circ \tilde{\mathcal{E}}_{M-1} \dots \circ \tilde{\mathcal{E}}_1
\end{equation}

Crucially, every active map $\mathcal{E}_k$ in our circuit model is structured around a physical Pauli generator $P_k$ (e.g., projecting onto $P_k$, rotating around $P_k$, or applying noise $P_k$). Conjugating the entire map by a cumulative Clifford frame $U$ distributes to the underlying generator. For example, the operator for a projective measurement $\Pi_k = \frac{1}{2}(I \pm P_k)$ maps to:
\begin{equation}
    U^\dagger \left[ \frac{1}{2}(I \pm P_k) \right] U = \frac{1}{2} \big( I \pm U^\dagger P_k U \big) = \frac{1}{2}(I \pm \tilde{P}_k)
\end{equation}
Similarly, a unitary rotation $e^{i\theta P_k}$ maps to $e^{i\theta \tilde{P}_k}$. The Clifford group normalizes the Pauli group, guaranteeing that the transformed generator $\tilde{P}_k$ is mapped to another valid Pauli string.

This collapse demonstrates a 1-to-1 mapping: executing local active operations dynamically amidst a shifting geometric frame ($\mathcal{E}_k$) is identical to composing a flat sequence of rewound operations ($\tilde{\mathcal{E}}_k$) parameterized by $t=0$ Paulis, evaluated within the static $t=0$ reference frame, followed by a single global geometric coordinate transformation $\mathcal{U}_M$ at the very end. This derivation relies on map identities and is independent of the initial state.

\subsection{The Front-End}

The Front-End compiler performs this rewinding mapping. It tracks the cumulative frame $U_k$ via stabilizer tableaus, absorbs the intermediate $C_k$ blocks, rewinds each active operation to find $\tilde{P}_k$, and evaluates the Pauli Extraction Map $\Gamma(\tilde{P}_k, \mathcal{T}^{(0)})$ against the initial computational tableau $\mathcal{T}^{(0)}$ to isolate the bitwise masks $(\omega, \mathbf{x}, \mathbf{z})$.

This sequence of extracted $t=0$ masks forms the HIR. Because deterministic Cliffords are geometrically absorbed by the tracking of $U_k$, their topological structure vanishes from the representation. However, because symplectic conjugation mathematically drops the continuous global phase (i.e., $U^\dagger P U$ is identical for $U$ and $e^{i\theta}U$), the Front-End explicitly extracts the continuous phase $\theta_U$ of each absorbed Clifford and accumulates it into a single, circuit-wide scalar $\theta_{\text{global}}$. The HIR is therefore a flat sequence of abstract active operations parameterized globally by this single scalar. Each operation type stores its extracted $t=0$ Pauli string alongside its semantic parameters:
\begin{itemize}
    \item \textbf{\texttt{HIR\_NON\_CLIFFORD}:} Represents a parameterized rotation expanded as a sum of Pauli terms. Stores the absolute Pauli operator and the relative complex weight $c$ of the target branch.
    \item \textbf{\texttt{HIR\_MEASURE}:} Represents a projective state collapse. Stores the absolute Pauli observable and a target integer index for the classical memory sequence $\mathcal{M}$.
    \item \textbf{\texttt{HIR\_NOISE}:} Represents a stochastic Pauli error channel. Stores the absolute Pauli error operator and the continuous physical probability $p_E$ of the error occurring.
    \item \textbf{\texttt{HIR\_FEEDBACK}:} Represents a classically controlled Pauli operation. Stores the absolute Pauli operator and the source integer index $k$ pointing to a previously recorded outcome $m^{(k)} \in \mathcal{M}$.
\end{itemize}

\subsection{The Middle-End Optimizer}

By expressing all operations in a unified $t=0$ coordinate system, the Middle-End Optimizer can evaluate circuit relationships globally without requiring iterative geometric frame updates.

To a skeptical reader, it might seem perilous to reorder or fuse active operations $A$ and $B$ that occur at different times, separated by intermediate operations, and within distinct geometric Clifford frames in the original circuit. However, we can justify this by proving that physical commutation is invariant beneath a geometric coordinate transformation, and by leveraging the algebraic properties of quantum map composition.

First, suppose $P_A$ and $P_B$ are physical Pauli generators evaluated in a localized cumulative frame $U$. Their rewound $t=0$ equivalents are $\tilde{P}_A = U^\dagger P_A U$ and $\tilde{P}_B = U^\dagger P_B U$. Expanding the commutator of these rewound operators yields:
\begin{align*}
    [\tilde{P}_A, \tilde{P}_B] &= \tilde{P}_A \tilde{P}_B - \tilde{P}_B \tilde{P}_A \\
    &= (U^\dagger P_A U)(U^\dagger P_B U) - (U^\dagger P_B U)(U^\dagger P_A U) \\
    &= U^\dagger (P_A P_B) U - U^\dagger (P_B P_A) U \\
    &= U^\dagger [P_A, P_B] U
\end{align*}
Because $U$ is a unitary matrix, $[\tilde{P}_A, \tilde{P}_B] = 0$ if and only if $[P_A, P_B] = 0$. This guarantees that operations commute in their true forward-moving runtime frame if and only if their extracted $t=0$ Pauli generators commute.

Second, this property allows the compiler to bridge the temporal distance between operations separated by \textit{other} active operations. Because Section 5.1 formalizes the circuit into a flat sequence of composed maps (e.g., $\dots \circ \tilde{\mathcal{E}}_B \circ \tilde{\mathcal{E}}_X \circ \tilde{\mathcal{E}}_A \circ \dots$), the compiler can reorder them pairwise. Because the rewound maps $\tilde{\mathcal{E}}$ rely on their $t=0$ Pauli generators, standard quantum mechanics dictates that adjacent maps commute ($\tilde{\mathcal{E}}_X \circ \tilde{\mathcal{E}}_A = \tilde{\mathcal{E}}_A \circ \tilde{\mathcal{E}}_X$) if and only if their underlying $t=0$ Paulis commute.

To evaluate this, the optimizer computes the symplectic inner product over $\mathbb{F}_2$ between the $t=0$ bitmasks of operation $A$ and an intervening operation $X$:
\begin{equation}
    (\mathbf{x}_A \cdot \mathbf{z}_X) \oplus (\mathbf{z}_A \cdot \mathbf{x}_X) \equiv 0 \pmod 2
\end{equation}
This single $\mathcal{O}(1)$ bitwise check determines commutativity. If it evaluates to zero, the compiler swaps their execution order, allowing operation $A$ to ``slide'' past $X$. Through successive commuting swaps (similar to a bubble sort), distant operations can become adjacent. Once adjacent, the optimizer compresses the circuit using the following rules:

\begin{itemize}
    \item \textbf{Algebraic Fusion:} If adjacent \texttt{HIR\_NON\_CLIFFORD} operations target the exact same topological Pauli axis ($\mathbf{x}_A = \mathbf{x}_B$ and $\mathbf{z}_A = \mathbf{z}_B$), they are fused and their relative complex weights combined. This can cancel the branch entirely or promote it to a geometric Clifford operation (e.g., $T \cdot T = e^{i\pi/4} S$). If promoted to a Clifford, the node is absorbed: the compiler explicitly conjugates all subsequent HIR nodes by this new Clifford to update their $\mathbf{x}$ and $\mathbf{z}$ bitmasks, and extracts the resulting mathematically generated scalar phase (e.g., $\pi/4$) to add to the global accumulator $\theta_{\text{global}}$.
    \item \textbf{Stochastic Barriers:} To preserve the correct chronological evaluation of the physical state equations derived in Section~\ref{sec:state_and_symplectic_determinism}, the \texttt{HIR\_MEASURE}, \texttt{HIR\_NOISE}, and \texttt{HIR\_FEEDBACK} nodes act as strict physical barriers. The optimizer will not reorder operations across these barriers if they anticommute, ensuring the physical observable eigenvalues and dynamic phase trajectories remain intact.
\end{itemize}

\subsection{The Compiler Back-End (Code Generation)}

Once the HIR is optimized in the global $t=0$ frame, the Compiler Back-End translates it back into the local, forward-moving geometric frame $\mathcal{T}^{(t)}$ required by the runtime state evolution rules.

The Back-End steps sequentially through the HIR, simulating the geometric state offline. For each operation, it projects the $t=0$ operator into the current intermediate frame $\mathcal{T}^{(t)}$. It performs the $\mathcal{O}(n^2)$ Symplectic Rowsums and Aaronson-Gottesman (AG) Pivots to resolve the geometric topology and extracts the physical, aligned payloads ($\omega', \mathbf{x}', \mathbf{z}', k_p$) required by the VM.

\textbf{Memory Pre-allocation:} Simultaneously, the Back-End tracks the active support mask $\mathbf{b}^{(t)}$ step-by-step. Because dimensionality expansion occurs deterministically, the compiler computes the peak active dimension reached during the entire circuit: $k_{\max} = \max_t |\mathbf{b}^{(t)}|$. Knowing this maximum active rank bounds the required runtime memory, allowing the Back-End to embed this limit $k_{\max}$---alongside the final global phase accumulator $\theta_{\text{global}}$---into the static header of the bytecode. This eliminates dynamic memory allocation during execution and avoids emitting $\mathcal{O}(G_{\text{clifford}})$ individual phase instructions into the payload.

\subsection{The Schrodinger VM and Opcodes}

The Schrodinger VM is a minimal execution environment that exclusively manages the probabilistic state variables: the classical memory sequence $\mathcal{M}$, the global phase tracking variable $\gamma$, the runtime parity trackers $\mathbf{p}_X, \mathbf{p}_Z$, and the complex amplitude array $a$.

At startup, the VM reads $k_{\max}$ and the global phase accumulator $\theta_{\text{global}}$ from the compiler payload header. It statically allocates the complex array $a$ to $2^{k_{\max}}$ entries. The array is initialized with zeros, setting $a_{\mathbf{0}} = 1$. It also initializes the global magnitude tracker using the accumulated phase: $\gamma^{(0)} = e^{i\theta_{\text{global}}}$. The VM then iterates through the opcodes emitted by the Back-End, executing the physical updates defined by the unified state evolution equations in Section~\ref{sec:state_and_symplectic_determinism}:

\begin{itemize}
    \item \textbf{\texttt{OP\_REINDEX}:} \\
    \textit{Payload:} Target index $i$, Source index $j$. \\
    \textit{Action:} Emitted when the Back-End performs an active-source rowsum offline. Instructs the VM to physically swap and reindex amplitudes in the array $a$ to satisfy the Amplitude Reindexing Map (Eq.~\ref{eq:reindexing}), assigning $a_{\boldsymbol{\beta}} \leftarrow a_{\boldsymbol{\beta} \oplus (\boldsymbol{\beta}_j \cdot \mathbf{e}_i)}$. The zero-cost dormant property guarantees this opcode is omitted for dormant sources.

    \item \textbf{\texttt{OP\_MIX}:} \\
    \textit{Payload:} Phase $\omega'$, masks $\mathbf{x}', \mathbf{z}'$, relative complex weight $c$. \\
    \textit{Action:} Evaluates the aligned dynamic phase $\Phi'(\boldsymbol{\alpha})$ against the current parities $\mathbf{p}_X, \mathbf{p}_Z$. Linearly mixes paired amplitudes $a_{\boldsymbol{\alpha}}$ and $a_{\boldsymbol{\alpha} \oplus \mathbf{x}'}$ utilizing the weight $c$. Updates the global magnitude tracker $\gamma$. Because the array is already allocated to the maximum dimension $2^{k_{\max}}$, structural expansion happens by updating the local active bounds and mixing with the pre-allocated zeros.

    \item \textbf{\texttt{OP\_MEAS\_RANDOM}:} (Maps to Case 3a) \\
    \textit{Payload:} Phase $\omega'$, pivot index $k_p$, aligned mask $\mathbf{z}'$. \\
    \textit{Action:} Samples $m \in \{0, 1\}$ using the absolute expectation $P_0=0.5$. Anchors the runtime parities $p_{X, k_p} \leftarrow p_{Z, k_p}$ and $p_{Z, k_p} \leftarrow m$. Appends $m$ to sequence $\mathcal{M}$. The complex array $a$ and global magnitude $\gamma$ are invariant.

    \item \textbf{\texttt{OP\_MEAS\_INTERFERE}:} (Maps to Case 3b) \\
    \textit{Payload:} Phase $\omega'$, pivot index $k_p$, aligned mask $\mathbf{z}'$. \\
    \textit{Action:} Evaluates linear branch interference across the active elements of the unnormalized array $a$ to calculate the structural weight $p_0$. Calculates absolute probability $P_0$ and samples outcome $m$. Folds the array $a$ over itself (halving the active logical dimension) via division-free addition. Defers normalization to the global tracker via $\gamma \leftarrow \gamma / \sqrt{2 P_m}$. Anchors parities $\mathbf{p}_X, \mathbf{p}_Z$ and appends $m$ to $\mathcal{M}$.

    \item \textbf{\texttt{OP\_MEAS\_DIAGONAL}:} (Maps to Case 3c) \\
    \textit{Payload:} Phase $\omega'$, pivot index $k_p$, aligned mask $\mathbf{z}'$. \\
    \textit{Action:} Evaluates the unnormalized structural weight $p_0$ of branches matching the deterministic phase constraint. Calculates absolute probability $P_0$ and samples outcome $m$. Filters the array $a$ by the valid boolean slice $v$ without division. Defers normalization to the global phase scalar via $\gamma \leftarrow \frac{\gamma}{\sqrt{P_m}} \cdot (-1)^{p_{X, k_p}^{\text{tmp}} \cdot v}$. Anchors parities $\mathbf{p}_X, \mathbf{p}_Z$ and appends $m$ to $\mathcal{M}$.

    \item \textbf{\texttt{OP\_MEAS\_STATIC}:} (Maps to Case 3d) \\
    \textit{Payload:} Phase $\omega$, static mask $\mathbf{z}$. \\
    \textit{Action:} Evaluates outcome $m$ against the current vacuum parities $\mathbf{p}_Z$. Appends $m$ to $\mathcal{M}$. The array $a$ and tracking scalar $\gamma$ are invariant.

    \item \textbf{\texttt{OP\_NOISE}:} \\
    \textit{Payload:} Phase $\omega$, masks $\mathbf{x}, \mathbf{z}$, error probability $p_E$. \\
    \textit{Action:} Queries the continuous random variable $\xi^{(t)}$ to generate the indicator $b = \mathbf{1}_{\xi^{(t)} < p_E}$. Updates global phase $\gamma$ via the indicator polynomial $\gamma \leftarrow \gamma \cdot \big[ (1-b) + b \cdot \omega (-1)^{\mathbf{z} \cdot \mathbf{p}_Z} \big]$ and shifts parities $\mathbf{p}_Z \leftarrow \mathbf{p}_Z \oplus (b \cdot \mathbf{x})$ and $\mathbf{p}_X \leftarrow \mathbf{p}_X \oplus (b \cdot \mathbf{z})$.

    \item \textbf{\texttt{OP\_FEEDBACK}:} \\
    \textit{Payload:} Phase $\omega$, masks $\mathbf{x}, \mathbf{z}$, classical condition index $k$. \\
    \textit{Action:} Queries the memory sequence $m^{(k)} \in \mathcal{M}$ to set $b=m^{(k)}$. Updates global phase via the indicator polynomial $\gamma \leftarrow \gamma \cdot \big[ (1-b) + b \cdot \omega (-1)^{\mathbf{z} \cdot \mathbf{p}_Z} \big]$ and shifts parities $\mathbf{p}_Z \leftarrow \mathbf{p}_Z \oplus (b \cdot \mathbf{x})$ and $\mathbf{p}_X \leftarrow \mathbf{p}_X \oplus (b \cdot \mathbf{z})$.
\end{itemize}

\section{Concrete 2-Qubit State Walkthroughs}
\label{app:concrete_walkthrough}

To ground the abstracted mathematical update rules, this section walks through the complete evaluation of a small 2-qubit state. We construct a single unnormalized ``Active Superposition'' state, and then use it as a static branching point to independently evaluate each of the four measurement subcases. In the presentation below, bitstrings are indexed with the lowest index on the left, e.g. the bit for qubit 1 is the leftmost bit in the string.

To maximize readability, we assume runtime parities $\mathbf{p}_X = \mathbf{0}, \mathbf{p}_Z = \mathbf{0}$ initially. Because the global norm relies on the Deferred Normalization logic, we explicitly track the global parameter $\gamma$ ensuring true absolute probabilities are evaluated accurately alongside the unnormalized structural components.

\subsection*{A.1 Initialization and Clifford Conjugation}
We initialize the state to the vacuum $\ket{00}$. The geometric frame starts in the canonical computational basis, the active branch mask is empty, and the tracking variables initialize to 1:
\begin{align*}
    \mathcal{S}^{(0)} &= \langle +Z_1, +Z_2 \rangle, \quad \mathcal{D}^{(0)} = \langle +X_1, +X_2 \rangle \\
    \mathbf{b} &= 00, \quad a_{00} = 1, \quad \gamma^{(0)} = 1.0
\end{align*}
We apply a Clifford Hadamard gate to the first qubit ($U = H_1$). By Clifford Conjugation rules, we apply $H_1 P H_1$ to every generator. The geometry updates, but the probabilistic tuple passes through unchanged:
\begin{align*}
    \mathcal{S}^{(1)} &= \langle +X_1, +Z_2 \rangle, \quad \mathcal{D}^{(1)} = \langle +Z_1, +X_2 \rangle \\
    \mathbf{b} &= 00, \quad a_{00} = 1, \quad \gamma^{(1)} = 1.0
\end{align*}

\subsection*{A.2 Non-Clifford Dimensionality Expansion}
We apply the non-Clifford rotation $T_1 \propto [I - i\tan(\frac{\pi}{8}) Z_1]$ to the state.
\begin{enumerate}
    \item \textbf{Extraction:} We evaluate the physical operator $Z_1$ against the current frame. Because $d_1 = +Z_1$, the map trivially yields $\mathbf{x} = 10, \mathbf{z} = 00$ with $\omega' = +1$.
    \item \textbf{Expansion:} Because $\mathbf{b} = 00$, the operator relies on dormant axes: $\mathbf{x}_{\text{dorm}} = 10$. We geometrically update the active rank to $\mathbf{b} \leftarrow 00 \vee 10 = 10$. The complex array expands, conceptually allocating padded bits: $a^{\text{pad}}_{00} = 1, a^{\text{pad}}_{10} = 0$. In practice, we've pre-allocated this space so no work is needed.
    \item \textbf{Mixing:} The dynamic phase evaluates uniformly: $\Phi'(\boldsymbol{\alpha}) = (+1)(-1)^0 = +1$ for all $\boldsymbol{\alpha}$. Mixing $a^{\text{pad}}$ against the geometric shift $\mathbf{x}'=10$ yields the unnormalized branch amplitudes:
    \begin{align*}
        a_{00} &= 1 \\
        a_{10} &= - i\tan(\tfrac{\pi}{8})
    \end{align*}
   The global scalar tracker updates as $\gamma \leftarrow 1.0 \cdot e^{i\pi/8} \cos(\tfrac{\pi}{8}) = e^{i\pi/8} \cos(\tfrac{\pi}{8})$.

    \textit{(Note: The true absolute state norm is preserved: $|\gamma|^2 \sum |a|^2 = \cos^2(\frac{\pi}{8}) [1 + \tan^2(\frac{\pi}{8})] = \cos^2(\frac{\pi}{8}) \sec^2(\frac{\pi}{8}) = 1.0$.)}
\end{enumerate}
In the standard computational basis our state is now $\frac{1}{\sqrt{2}}\left(\ket{00} + e^{i\pi/4}\ket{10}\right)$.

\noindent \textit{(We now have an unnormalized active mask $\mathbf{b}=10$. We branch from this specific state to independently demonstrate the four distinct measurement scenarios.)}


\subsection*{A.3 Case 3a: Random Pivot / Vacuum Collapse}
\textit{Action:} Measure $P = Z_1 X_2$.
\begin{enumerate}
    \item \textbf{Extraction:} In our frame $\mathcal{T}^{(1)}$, $d_1 = +Z_1$ and $d_2 = +X_2$, so $P = +d_1 d_2$. The extraction map yields $\omega' = +1$, $\mathbf{x} = 11$, $\mathbf{z} = 00$.
    \item \textbf{Intersection:} With $\mathbf{b} = 10$, we have $\mathbf{x}_{\text{act}} = 10$ and $\mathbf{x}_{\text{dorm}} = 01$. The measurement acts outside the active superposition. We select the dormant bit $k_p=2$ as the pivot.
    \item \textbf{Rowsum:} We apply $\Lambda_{\text{rowsum}}(1, 2)$ to the geometry:
    \begin{align*}
        s_1 &\leftarrow s_1 s_2 = +X_1 Z_2 \\
        d_2 &\leftarrow d_2 d_1 = +Z_1 X_2
    \end{align*}
    The \textit{Zero-Cost Dormant Property} ensures our complex array $a$ is untouched by this compression.
    \item \textbf{Pivot:} We now recompose $P$, which is now $d_2$ in the new geometry. We have support only over $a_{00}$ and $a_{10}$, so there is no interference on this second qubit. The outcome is thus a 50-50 coin-flip. After we get the outcome $m$, we apply $\Lambda_{\text{pivot}}$ to replace $s_2 \leftarrow P = Z_1 X_2$, and demote the old $s_2 = Z_2$ to the new $d_2$. $d_1$ remains $Z_1$.
\end{enumerate}

\subsection*{A.4 Case 3b: Interfering Shrink}
\textit{Action:} Measure $P = Z_1$.
\begin{enumerate}
    \item \textbf{Extraction:} Because $d_1 = +Z_1$, extraction yields $\omega' = +1$, $\mathbf{x} = 10$, $\mathbf{z} = 00$.
    \item \textbf{Intersection:} With $\mathbf{b} = 10$, we partition $\mathbf{x}_{\text{act}} = 10$ and $\mathbf{x}_{\text{dorm}} = 00$. The measurement acts linearly within the active superposition. We select the active bit $k_p=1$ as the pivot.
    \item \textbf{Update:} The active dimension predictably halves: $\mathbf{b} \leftarrow 00$. The VM folds the state over the 1st geometric axis, calculating the relative unnormalized structural weight $p_0$:
    \begin{equation*}
        p_0 = \tfrac{1}{2} \big| a_{00} + a_{10} \big|^2 = \tfrac{1}{2} \big| 1 - i\tan(\tfrac{\pi}{8}) \big|^2 = \tfrac{1}{2} \sec^2(\tfrac{\pi}{8}) = \frac{1}{2\cos^2(\frac{\pi}{8})}
    \end{equation*}
    To evaluate the true absolute probability $P_0$ for sampling, the VM evaluates $p_0$ against the global magnitude ($|\gamma|^2 = \cos^2(\frac{\pi}{8})$):
    \begin{equation*}
        P_0 = p_0 \cdot |\gamma|^2 = \left( \frac{1}{2\cos^2(\frac{\pi}{8})} \right) \cdot \cos^2(\tfrac{\pi}{8}) = 0.5
    \end{equation*}
    \textit{(Note: This matches the physical textbook expectation for measuring $Z$ on $T\ket{+}$, avoiding divisions over the runtime array).}

    Observing outcome $m$, the single remaining branch is derived via division-free addition:
    \begin{equation*}
        a_{00} \leftarrow a_{00} + (-1)^m a_{10} = 1 - i(-1)^m\tan(\tfrac{\pi}{8})
    \end{equation*}
    To preserve the physical state norm post-measurement, Deferred Normalization updates the global magnitude instead of the array:
    \begin{equation*}
        \gamma \leftarrow \frac{\gamma}{\sqrt{2P_m}} = \frac{e^{i\pi/8} \cos(\pi/8)}{\sqrt{1}} = e^{i\pi/8} \cos(\tfrac{\pi}{8})
    \end{equation*}
    The geometric pivot operation subsequently demotes $k_p=1$ to anchor the new probabilistic vacuum.
\end{enumerate}

\subsection*{A.5 Case 3c: Diagonal Shrink}
\textit{Action:} Measure $P = X_1$.
\begin{enumerate}
    \item \textbf{Extraction:} Because $s_1 = +X_1$, extraction maps $\omega' = +1$, $\mathbf{x} = 00$, $\mathbf{z} = 10$.
    \item \textbf{Intersection:} $\mathbf{x} = \mathbf{0}$, meaning there is no transversal mixing shift. But because $\mathbf{b}=10$, we find $\mathbf{z}_{\text{act}} = 10$. The measurement acts diagonally across the active superposition.
    \item \textbf{Update:} The dynamic phase evaluates as $\Phi'(\boldsymbol{\alpha}_{00}) = +1$ and $\Phi'(\boldsymbol{\alpha}_{10}) = -1$. The unnormalized structural expectation evaluates to $p_0 = |a_{00}|^2 = 1.0$.
    \item \textbf{Filter:} Integrating the scalar, the true absolute probability evaluates to $P_0 = p_0 \cdot |\gamma|^2 = 1.0 \cdot \cos^2(\frac{\pi}{8}) \approx 85.3\%$ (matching the physical reality of measuring $X$ on $T\ket{+}$).

    Assuming the VM samples outcome $m=0$, the active dimension correctly halves ($\mathbf{b} \leftarrow 00$). The valid boolean slice $v=0$ filters the array without division: we discard $a_{10}$ and keep $a_{00} = 1$. Evaluating Deferred Normalization scales the physical state limits:
    \begin{equation*}
        \gamma \leftarrow \frac{\gamma}{\sqrt{P_m}} = \frac{e^{i\pi/8} \cos(\pi/8)}{\cos(\pi/8)} = e^{i\pi/8}
